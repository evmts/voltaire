---
title: Verify NFT Authenticity
description: Verify an NFT is authentic by checking contract address, ownership, and mint history.
---

## Goal

Given an NFT contract address and token ID, verify authenticity by confirming ownership and tracing mint origin.

Return as:

```
AUTHENTIC: true/false
OWNER: 0x...
MINTED_AT_BLOCK: 12345678
```

## Example (TypeScript)

```ts
import { Contract } from '@voltaire/contract'
import * as Address from '@voltaire/primitives/Address'

const ERC721_ABI = [
  {
    type: 'function',
    name: 'ownerOf',
    stateMutability: 'view',
    inputs: [{ name: 'tokenId', type: 'uint256' }],
    outputs: [{ name: '', type: 'address' }]
  },
  {
    type: 'event',
    name: 'Transfer',
    inputs: [
      { name: 'from', type: 'address', indexed: true },
      { name: 'to', type: 'address', indexed: true },
      { name: 'tokenId', type: 'uint256', indexed: true }
    ]
  }
] as const

// Known authentic contract addresses
const VERIFIED_CONTRACTS = new Set([
  '0xBC4CA0EdA7647A8aB7C2061c2E118A18a936f13D', // BAYC
  '0xb47e3cd837dDF8e4c57F05d70Ab865de6e193BBB', // CryptoPunks
  '0x60E4d786628Fea6478F785A6d7e704777c86a7c6', // MAYC
])

async function verifyNFT(contractAddress: string, tokenId: bigint) {
  // 1. Check contract is verified
  const checksummed = Address.toChecksummed(contractAddress)
  if (!VERIFIED_CONTRACTS.has(checksummed)) {
    return { authentic: false, reason: 'Unknown contract' }
  }

  const nft = Contract({
    address: contractAddress,
    abi: ERC721_ABI,
    provider
  })

  // 2. Verify token exists by checking ownership
  let owner: string
  try {
    owner = await nft.read.ownerOf(tokenId)
  } catch {
    return { authentic: false, reason: 'Token does not exist' }
  }

  // 3. Find mint event (Transfer from 0x0)
  const ZERO_ADDRESS = '0x0000000000000000000000000000000000000000'
  let mintBlock: bigint | null = null

  for await (const log of nft.events.Transfer({
    from: ZERO_ADDRESS,
    tokenId
  })) {
    mintBlock = log.blockNumber
    break // First Transfer from zero is the mint
  }

  console.log(`AUTHENTIC: true`)
  console.log(`OWNER: ${owner}`)
  console.log(`MINTED_AT_BLOCK: ${mintBlock}`)

  return { authentic: true, owner, mintBlock }
}

await verifyNFT('0xBC4CA0EdA7647A8aB7C2061c2E118A18a936f13D', 1234n)
```
