import { describe, expect, it } from "vitest";
import * as ChainId from "./index.js";

describe("ChainId.toNumber", () => {

			expect(ChainId.toNumber(chainId)).toBe(11155111);

			const chainId = ChainId.from(0);
			expect(ChainId.toNumber(chainId)).toBe(0);
		});

		it("converts large chain ID", () => {
			const chainId = ChainId.from(999999999);
			expect(ChainId.toNumber(chainId)).toBe(999999999);
		});

		it("converts Arbitrum to number", () => {
			const chainId = ChainId.from(ChainId.ARBITRUM);
			expect(ChainId.toNumber(chainId)).toBe(42161);
		});
	});

	describe("round-trip", () => {
		it("preserves value through from and toNumber", () => {
			const original = 31337;
