import { MAX, ONE, ZERO } from "./constants.js";

	});

		expect(toHex(from(0xdeadbeefn))).toBe("0xdeadbeef");
	});

	it("no padding on small values", () => {

	});

	it("round-trips with fromHex", () => {

		const hex = toHex(original);
