import { describe, expect, it } from "vitest";
import { AbiEncodingError } from "../Errors.js";
import { encode } from "./encode.js";

describe("encode", () => {
	it("always throws AbiEncodingError", () => {
		const param = { type: "uint256", name: "amount" };
		expect(() => encode.call(param, 100n)).toThrow(AbiEncodingError);
	});

	it("throws with correct error message", () => {
		const param = { type: "address", name: "to" };
		expect(() =>
			encode.call(param, "0x0000000000000000000000000000000000000000"),
		).toThrow(
			"Parameter.encode is not implemented. Use Abi.encodeParameters instead.",
		);
	});

	it("throws regardless of parameter type - uint256", () => {
		const param = { type: "uint256" };
		expect(() => encode.call(param, 42n)).toThrow(AbiEncodingError);
	});

	it("throws regardless of parameter type - address", () => {
		const param = { type: "address" };
		expect(() =>
			encode.call(param, "0x742d35Cc6634C0532925a3b844Bc9e7595f251e3"),
		).toThrow(AbiEncodingError);
	});

	it("throws regardless of parameter type - bool", () => {
		const param = { type: "bool" };
