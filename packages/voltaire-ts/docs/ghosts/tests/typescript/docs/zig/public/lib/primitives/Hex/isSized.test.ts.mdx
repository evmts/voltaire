import { describe, expect, it } from "vitest";
import type { HexType } from "./HexType.js";
import { isSized } from "./isSized.js";

describe("isSized", () => {
	it("returns true for correct size", () => {
		expect(isSized("0x" as HexType, 0)).toBe(true);
		expect(isSized("0x00" as HexType, 1)).toBe(true);
		expect(isSized("0x1234" as HexType, 2)).toBe(true);
		expect(isSized("0x123456" as HexType, 3)).toBe(true);
		expect(isSized("0x12345678" as HexType, 4)).toBe(true);
	});

	it("returns false for incorrect size", () => {
		expect(isSized("0x1234" as HexType, 1)).toBe(false);
		expect(isSized("0x1234" as HexType, 3)).toBe(false);
		expect(isSized("0x" as HexType, 1)).toBe(false);
		expect(isSized("0x00" as HexType, 0)).toBe(false);
	});

	it("handles large sizes", () => {
		const hex = `0x${"00".repeat(32)}` as HexType;
		expect(isSized(hex, 32)).toBe(true);
		expect(isSized(hex, 31)).toBe(false);
		expect(isSized(hex, 33)).toBe(false);
	});

	it("handles address size (20 bytes)", () => {
		const address = `0x${"00".repeat(20)}` as HexType;
		expect(isSized(address, 20)).toBe(true);
		expect(isSized(address, 19)).toBe(false);
	});

	it("handles hash size (32 bytes)", () => {
		const hash = `0x${"00".repeat(32)}` as HexType;
		expect(isSized(hash, 32)).toBe(true);
		expect(isSized(hash, 31)).toBe(false);
	});

	it("works with zero target size", () => {
		expect(isSized("0x" as HexType, 0)).toBe(true);
		expect(isSized("0x00" as HexType, 0)).toBe(false);
	});
});
