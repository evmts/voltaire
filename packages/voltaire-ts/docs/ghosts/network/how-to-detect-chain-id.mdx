# Detect Chain ID

Use `eth_chainId` to identify the network.

```typescript
import { createJsonRpcProvider } from '@tevm/voltaire/jsonrpc';
import * as Rpc from '@tevm/voltaire/jsonrpc';

const provider = createJsonRpcProvider('https://eth.llamarpc.com');

const chainIdHex = await provider.request(Rpc.Eth.ChainIdRequest());
const chainId = parseInt(chainIdHex, 16);

console.log('Chain ID:', chainId);
// 1 (Ethereum mainnet)
```

## Network Detection Helper

```typescript
const networks: Record<number, string> = {
  1: 'Ethereum Mainnet',
  11155111: 'Sepolia',
  17000: 'Holesky',
  137: 'Polygon',
  10: 'Optimism',
  42161: 'Arbitrum One',
  8453: 'Base',
};

async function detectNetwork(provider) {
  const chainIdHex = await provider.request(Rpc.Eth.ChainIdRequest());
  const chainId = parseInt(chainIdHex, 16);
  return networks[chainId] || `Unknown (${chainId})`;
}
```
