1//! SHA-256 hash function.
2
3/// Compute SHA-256 hash of input data.
4///
5/// # Examples
6///
7/// ```rust
8/// use voltaire::crypto::sha256;
9///
10/// let hash = sha256(b"hello");
11/// ```
12#[cfg(feature = "native")]
13pub fn sha256(data: &[u8]) -> [u8; 32] {
14    crate::ffi::sha256(data)
15}
16
17/// Pure Rust SHA-256 implementation.
18#[cfg(not(feature = "native"))]
19pub fn sha256(data: &[u8]) -> [u8; 32] {
20    // Pure Rust SHA-256 implementation
21    const K: [u32; 64] = [
22        0x428a2f98, 0x71374491, 0xb5c0fbcf, 0xe9b5dba5,
23        0x3956c25b, 0x59f111f1, 0x923f82a4, 0xab1c5ed5,
24        0xd807aa98, 0x12835b01, 0x243185be, 0x550c7dc3,
25        0x72be5d74, 0x80deb1fe, 0x9bdc06a7, 0xc19bf174,
26        0xe49b69c1, 0xefbe4786, 0x0fc19dc6, 0x240ca1cc,
27        0x2de92c6f, 0x4a7484aa, 0x5cb0a9dc, 0x76f988da,
28        0x983e5152, 0xa831c66d, 0xb00327c8, 0xbf597fc7,
29        0xc6e00bf3, 0xd5a79147, 0x06ca6351, 0x14292967,
30        0x27b70a85, 0x2e1b2138, 0x4d2c6dfc, 0x53380d13,
31        0x650a7354, 0x766a0abb, 0x81c2c92e, 0x92722c85,
32        0xa2bfe8a1, 0xa81a664b, 0xc24b8b70, 0xc76c51a3,
33        0xd192e819, 0xd6990624, 0xf40e3585, 0x106aa070,
34        0x19a4c116, 0x1e376c08, 0x2748774c, 0x34b0bcb5,
35        0x391c0cb3, 0x4ed8aa4a, 0x5b9cca4f, 0x682e6ff3,
36        0x748f82ee, 0x78a5636f, 0x84c87814, 0x8cc70208,
37        0x90befffa, 0xa4506ceb, 0xbef9a3f7, 0xc67178f2,
38    ];
39
40    let mut h: [u32; 8] = [
41        0x6a09e667, 0xbb67ae85, 0x3c6ef372, 0xa54ff53a,
42        0x510e527f, 0x9b05688c, 0x1f83d9ab, 0x5be0cd19,
43    ];
44
45    // Pre-processing: add padding
46    let ml = (data.len() as u64) * 8; // message length in bits
47    let mut padded = data.to_vec();
48    padded.push(0x80);
49
50    while (padded.len() % 64) != 56 {
51        padded.push(0x00);
52    }
53
54    padded.extend_from_slice(&ml.to_be_bytes());
55
56    // Process each 64-byte chunk
57    for chunk in padded.chunks_exact(64) {
58        let mut w = [0u32; 64];
59
60        for (i, word) in chunk.chunks_exact(4).enumerate() {
61            w[i] = u32::from_be_bytes(word.try_into().unwrap());
62        }
63
64        for i in 16..64 {
65            let s0 = w[i-15].rotate_right(7) ^ w[i-15].rotate_right(18) ^ (w[i-15] >> 3);
66            let s1 = w[i-2].rotate_right(17) ^ w[i-2].rotate_right(19) ^ (w[i-2] >> 10);
67            w[i] = w[i-16].wrapping_add(s0).wrapping_add(w[i-7]).wrapping_add(s1);
68        }
69
70        let mut a = h[0];
71        let mut b = h[1];
72        let mut c = h[2];
73        let mut d = h[3];
74        let mut e = h[4];
75        let mut f = h[5];
76        let mut g = h[6];
77        let mut hh = h[7];
78
79        for i in 0..64 {
80            let s1 = e.rotate_right(6) ^ e.rotate_right(11) ^ e.rotate_right(25);
81            let ch = (e & f) ^ ((!e) & g);
82            let temp1 = hh.wrapping_add(s1).wrapping_add(ch).wrapping_add(K[i]).wrapping_add(w[i]);
83            let s0 = a.rotate_right(2) ^ a.rotate_right(13) ^ a.rotate_right(22);
84            let maj = (a & b) ^ (a & c) ^ (b & c);
85            let temp2 = s0.wrapping_add(maj);
86
87            hh = g;
88            g = f;
89            f = e;
90            e = d.wrapping_add(temp1);
91            d = c;
92            c = b;
93            b = a;
94            a = temp1.wrapping_add(temp2);
95        }
96
97        h[0] = h[0].wrapping_add(a);
98        h[1] = h[1].wrapping_add(b);
99        h[2] = h[2].wrapping_add(c);
100        h[3] = h[3].wrapping_add(d);
101        h[4] = h[4].wrapping_add(e);
102        h[5] = h[5].wrapping_add(f);
103        h[6] = h[6].wrapping_add(g);
104        h[7] = h[7].wrapping_add(hh);
105    }
106
107    let mut output = [0u8; 32];
108    for (i, word) in h.iter().enumerate() {
109        output[i*4..(i+1)*4].copy_from_slice(&word.to_be_bytes());
110    }
111    output
112}
