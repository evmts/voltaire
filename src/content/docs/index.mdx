## Quick Example

```tsx live
import { useState } from 'react';

export default function VoltaireDemo() {
  const [address, setAddress] = useState('0xa0cf798816d4b9b9866b5330eea46a18382f251e');

  return (
    <div style={{ padding: '1rem', border: '1px solid #ccc', borderRadius: '8px' }}>
      <h3>Try it yourself</h3>
      <input
        type="text"
        value={address}
        onChange={(e) => setAddress(e.target.value)}
        style={{ width: '100%', padding: '0.5rem', marginBottom: '1rem' }}
      />
      <pre style={{ background: '#f5f5f5', padding: '1rem', borderRadius: '4px' }}>
        <code>{`import { Address } from "@tevm/voltaire";

const addr = new Address("${address}");
const checksum = Address.toChecksumHex.call(addr);
const isZero = Address.isZero.call(addr);`}</code>
      </pre>
      <p style={{ marginTop: '1rem' }}>
        <strong>Note:</strong> Install @tevm/voltaire to run this code
      </p>
    </div>
  );
}
```

## Installation

```bash
npm install @tevm/voltaire
```

```bash
bun add @tevm/voltaire
```

```bash
pnpm install @tevm/voltaire
```

Or use the [Zig API](https://github.com/evmts/voltaire/blob/main/docs/ZIG_API.md):

```bash
# Install specific version (recommended)
zig fetch --save https://github.com/evmts/voltaire/archive/refs/tags/v0.1.0.tar.gz

# Install latest from main branch
zig fetch --save git+https://github.com/evmts/voltaire
```
