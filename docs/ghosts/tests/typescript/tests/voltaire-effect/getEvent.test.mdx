## Gets Transfer event

Retrieves event definition by name from ABI.

```typescript
import { describe, expect, it } from "@effect/vitest"
import * as Effect from "effect/Effect"
import * as S from "effect/Schema"
import * as Abi from "voltaire-effect/primitives/Abi"

const erc20Abi = S.decodeUnknownSync(Abi.fromArray)([
  {
    type: "function",
    name: "transfer",
    stateMutability: "nonpayable",
    inputs: [
      { name: "to", type: "address" },
      { name: "amount", type: "uint256" },
    ],
    outputs: [{ type: "bool" }],
  },
  {
    type: "event",
    name: "Transfer",
    inputs: [
      { name: "from", type: "address", indexed: true },
      { name: "to", type: "address", indexed: true },
      { name: "value", type: "uint256", indexed: false },
    ],
  },
  {
    type: "event",
    name: "Approval",
    inputs: [
      { name: "owner", type: "address", indexed: true },
      { name: "spender", type: "address", indexed: true },
      { name: "value", type: "uint256", indexed: false },
    ],
  },
])

it.effect("gets Transfer event", () =>
  Effect.gen(function* () {
    const evt = yield* Abi.getEvent(erc20Abi, "Transfer")
    expect(evt.name).toBe("Transfer")
    expect(evt.type).toBe("event")
    expect(evt.inputs.length).toBe(3)
  }),
)
```

## Gets event with indexed inputs

Retrieves event and verifies indexed input flags.

```typescript
it.effect("gets event with indexed inputs", () =>
  Effect.gen(function* () {
    const evt = yield* Abi.getEvent(erc20Abi, "Transfer")
    const indexed = evt.inputs.filter((i) => i.indexed)
    expect(indexed.length).toBe(2)
  }),
)
```

## Fails for unknown event

Returns failure when event name doesn't exist in ABI.

```typescript
import * as Exit from "effect/Exit"

it("fails for unknown event", async () => {
  const exit = await Effect.runPromiseExit(
    Abi.getEvent(erc20Abi, "UnknownEvent"),
  )
  expect(Exit.isFailure(exit)).toBe(true)
})
```

## Fails for function name

Returns failure when name matches function instead of event.

```typescript
it("fails for function name (not an event)", async () => {
  const exit = await Effect.runPromiseExit(Abi.getEvent(erc20Abi, "transfer"))
  expect(Exit.isFailure(exit)).toBe(true)
})
```

## Can catch error with Effect.catchAll

Demonstrates error handling with Effect.catchAll.

```typescript
it.effect("can catch error with Effect.catchAll", () =>
  Effect.gen(function* () {
    const result = yield* Abi.getEvent(erc20Abi, "unknown").pipe(
      Effect.catchAll(() =>
        Effect.succeed({ name: "fallback", type: "event" as const }),
      ),
    )
    expect(result.name).toBe("fallback")
  }),
)
```
