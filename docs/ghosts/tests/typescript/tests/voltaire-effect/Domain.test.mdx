## Decodes valid domain object

Demonstrates decoding an EIP-712 domain structure.

```typescript
import { describe, expect, it } from "@effect/vitest"
import * as Schema from "effect/Schema"
import * as Domain from "voltaire-effect/primitives/Domain"

it("decodes valid domain object", () => {
  const input = {
    name: "MyDApp",
    version: "1",
    chainId: 1n,
  }
  const result = Schema.decodeSync(Domain.Struct)(input)
  expect(result.name).toBe("MyDApp")
  expect(result.version).toBe("1")
})
```

## Decodes with verifying contract

Demonstrates domain with verifyingContract field.

```typescript
import * as Schema from "effect/Schema"
import * as Domain from "voltaire-effect/primitives/Domain"

it("decodes with verifying contract", () => {
  const input = {
    name: "MyDApp",
    version: "1",
    chainId: 1n,
    verifyingContract: `0x${"00".repeat(20)}`,
  }
  const result = Schema.decodeSync(Domain.Struct)(input)
  expect(result.name).toBe("MyDApp")
})
```

## Encodes back to input format

Demonstrates round-trip encoding.

```typescript
import * as Schema from "effect/Schema"
import * as Domain from "voltaire-effect/primitives/Domain"

it("encodes back to input format", () => {
  const input = {
    name: "MyDApp",
    version: "1",
    chainId: 1n,
  }
  const domain = Schema.decodeSync(Domain.Struct)(input)
  const encoded = Schema.encodeSync(Domain.Struct)(domain)
  expect(encoded.name).toBe("MyDApp")
  expect(encoded.version).toBe("1")
})
```

## toHash computes domain separator

Demonstrates computing the EIP-712 domain separator hash.

```typescript
import { hash as keccak256 } from "@tevm/voltaire/Keccak256"
import * as Schema from "effect/Schema"
import * as Domain from "voltaire-effect/primitives/Domain"

it("toHash computes domain separator hash", () => {
  const input = {
    name: "MyDApp",
    version: "1",
    chainId: 1n,
  }
  const domain = Schema.decodeSync(Domain.Struct)(input)
  const hash = Domain.toHash(domain, { keccak256 })
  expect(hash).toBeInstanceOf(Uint8Array)
  expect(hash.length).toBe(32)
})
```

## getEIP712DomainType returns domain type fields

Demonstrates extracting EIP-712 domain type definition.

```typescript
import * as Schema from "effect/Schema"
import * as Domain from "voltaire-effect/primitives/Domain"

it("getEIP712DomainType returns domain type fields", () => {
  const input = {
    name: "MyDApp",
    version: "1",
    chainId: 1n,
  }
  const domain = Schema.decodeSync(Domain.Struct)(input)
  const fields = Domain.getEIP712DomainType(domain)
  expect(Array.isArray(fields)).toBe(true)
  expect(fields.some((f) => f.name === "name")).toBe(true)
})
```
