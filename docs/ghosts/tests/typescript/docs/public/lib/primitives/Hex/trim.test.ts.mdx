import { describe, expect, it } from "vitest";
import type { HexType } from "./HexType.js";
import { trim } from "./trim.js";

describe("trim", () => {
	it("removes leading zeros", () => {
		expect(trim("0x00001234" as HexType)).toBe("0x1234");
		expect(trim("0x0000abcd" as HexType)).toBe("0xabcd");
	});

	it("removes all leading zeros", () => {
		expect(trim("0x0000" as HexType)).toBe("0x");
		expect(trim("0x00000000" as HexType)).toBe("0x");
	});

	it("does not trim non-zero values", () => {
		expect(trim("0x1234" as HexType)).toBe("0x1234");
		expect(trim("0xabcd" as HexType)).toBe("0xabcd");
	});

	it("does not trim zeros in middle or end", () => {
		expect(trim("0x120034" as HexType)).toBe("0x120034");
		expect(trim("0x123400" as HexType)).toBe("0x123400");
	});

	it("trims empty hex", () => {
		expect(trim("0x" as HexType)).toBe("0x");
	});

	it("trims single zero byte", () => {
		expect(trim("0x00" as HexType)).toBe("0x");
	});

	it("preserves single non-zero byte", () => {
		expect(trim("0xff" as HexType)).toBe("0xff");
		expect(trim("0x01" as HexType)).toBe("0x01");
	});

	it("handles odd-length leading zeros", () => {
		expect(trim("0x001234" as HexType)).toBe("0x1234");
	});

	it("converts to lowercase when trimming (via fromBytes)", () => {
		expect(trim("0x0000ABCD" as HexType)).toBe("0xabcd");
		expect(trim("0x0000abcd" as HexType)).toBe("0xabcd");
	});

	it("trims many leading zeros", () => {
		const hex = `0x${"00".repeat(50)}1234` as HexType;
		expect(trim(hex)).toBe("0x1234");
	});

	it("handles zero value", () => {
		expect(trim("0x00" as HexType)).toBe("0x");
		expect(trim("0x0000000000000000" as HexType)).toBe("0x");
	});
});
