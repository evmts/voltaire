import { describe, expect, it } from "vitest";
import { fromBigInt } from "./fromBigInt.js";

describe("Bytes.fromBigInt", () => {
	it("converts zero", () => {
		expect(fromBigInt(0n)).toEqual(new Uint8Array([0]));
	});

	it("converts small values", () => {
		expect(fromBigInt(1n)).toEqual(new Uint8Array([1]));
		expect(fromBigInt(255n)).toEqual(new Uint8Array([255]));
		expect(fromBigInt(256n)).toEqual(new Uint8Array([1, 0]));
	});

	it("converts with size padding", () => {
		expect(fromBigInt(1n, 4)).toEqual(new Uint8Array([0, 0, 0, 1]));
		expect(fromBigInt(0x1234n, 4)).toEqual(new Uint8Array([0, 0, 0x12, 0x34]));
	});

	it("throws when value exceeds size", () => {
		expect(() => fromBigInt(256n, 1)).toThrow(/requires 2 bytes/);
	});

	it("throws on negative values", () => {
		expect(() => fromBigInt(-1n)).toThrow(/must be non-negative/);
	});

	it("handles large values (256 bits)", () => {
		const maxU256 = 2n ** 256n - 1n;
		const result = fromBigInt(maxU256);
		expect(result.length).toBe(32);
		expect(result.every((b) => b === 0xff)).toBe(true);
	});
});
