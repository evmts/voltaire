			yParity: 0,

		const signingHash = TransactionEIP7702.getSigningHash(unsignedTx);
		const signature = Secp256k1.sign(signingHash, privateKey);

		const signedTx: TransactionEIP7702Type = {

			yParity: signature.v - 27,
