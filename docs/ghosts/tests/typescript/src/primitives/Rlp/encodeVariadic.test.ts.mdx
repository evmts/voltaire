import { decode } from "./decode.js";
import { encodeVariadic } from "./encodeVariadic.js";
import { toRaw } from "./toRaw.js";

describe("Rlp.encodeVariadic", () => {
	it("encodes no arguments as empty list", () => {
		const result = encodeVariadic();
		const decoded = decode(result);
		const raw = toRaw(decoded.data);
		expect(Array.isArray(raw)).toBe(true);
		expect(raw).toHaveLength(0);
	});

	it("encodes single argument", () => {
		const result = encodeVariadic(new Uint8Array([1, 2, 3]));

		const raw = toRaw(decoded.data);
		expect(Array.isArray(raw)).toBe(true);
		expect(raw).toHaveLength(1);

	});

		const result = encodeVariadic(
