          const sandboxes = {
            address: 'https://codesandbox.io/embed/wclsxd?view=editor+%2B+preview&module=%2Fsrc%2Findex.mjs&hidenavigation=1',
            keccak: 'https://codesandbox.io/embed/wclsxd?view=editor+%2B+preview&module=%2Fsrc%2Findex.mjs&hidenavigation=1',
            transaction: 'https://codesandbox.io/embed/wclsxd?view=editor+%2B+preview&module=%2Fsrc%2Findex.mjs&hidenavigation=1',
            abi: 'https://codesandbox.io/embed/wclsxd?view=editor+%2B+preview&module=%2Fsrc%2Findex.mjs&hidenavigation=1'
          };

          let currentSandbox = 'address';

          function init() {
            const iframe = document.getElementById('voltaire-sandbox');
            const tabs = document.querySelectorAll('.sandbox-tab');

            if (!iframe || tabs.length === 0) {
              setTimeout(init, 50);
              return;
            }

            function getTheme() {
              const storedDarkMode = localStorage.getItem('isDarkMode');
              return storedDarkMode === 'true' ? 'dark' : 'light';
            }

            function updateIframe(sandboxKey) {
              const baseUrl = sandboxes[sandboxKey];
              iframe.src = baseUrl + '&theme=' + getTheme();
              currentSandbox = sandboxKey;
            }

            function setActiveTab(activeTab) {
              tabs.forEach(tab => {
                const isActive = tab === activeTab;
                tab.style.background = isActive ? 'linear-gradient(135deg, #E6A23C 0%, #A0522D 100%)' : 'transparent';
                tab.style.color = isActive ? 'white' : 'currentColor';
                tab.style.borderColor = isActive ? 'rgba(212, 147, 30, 0.5)' : 'rgba(128, 128, 128, 0.3)';
              });
            }

            tabs.forEach(tab => {
              tab.addEventListener('click', () => {
                const sandboxKey = tab.getAttribute('data-sandbox');
                updateIframe(sandboxKey);
                setActiveTab(tab);
              });
            });

            updateIframe('address');

            const observer = new MutationObserver(() => {
              const theme = getTheme();
              const currentTheme = iframe.src.includes('theme=light') ? 'light' : 'dark';
              if (currentTheme !== theme) {
                updateIframe(currentSandbox);
              }
            });

            observer.observe(document.documentElement, {
              attributes: true,
              attributeFilter: ['class', 'data-theme']
            });
          }

          if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', init);
          } else {
            init();
          }
        })();
      `}}></script>
