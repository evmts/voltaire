## Signature

```typescript
Blob.isValid(value: unknown): boolean
```

Validate that value is exactly 131,072 bytes. Used to check blob size before operations.

### Parameters

- **value** `unknown` - Value to validate

### Returns

`boolean` - `true` if value is Uint8Array with length 131,072, `false` otherwise

### Example

```typescript
import { Blob } from '@tevm/voltaire';

const validBlob = new Uint8Array(131072);
console.log(Blob.isValid(validBlob)); // true

const tooSmall = new Uint8Array(1000);
console.log(Blob.isValid(tooSmall)); // false

const tooLarge = new Uint8Array(200000);
console.log(Blob.isValid(tooLarge)); // false

const notArray = "not a blob";
console.log(Blob.isValid(notArray)); // false
```

### Example: Validation Guard

```typescript
import { Blob } from '@tevm/voltaire';

function processBlob(data: Uint8Array) {
  if (!Blob.isValid(data)) {
    throw new Error(`Invalid blob size: ${data.length} (expected ${Blob.SIZE})`);
  }

  // Safe to use as blob
  const commitment = Blob.toCommitment(data);
  return commitment;
}
```

## Usage

### Before Blob Operations

```typescript
import { Blob } from '@tevm/voltaire';

// Validate before expensive operations
if (Blob.isValid(data)) {
  const commitment = Blob.toCommitment(data);
  const proof = Blob.toProof(data);
} else {
  // Encode data into blob format
  const blob = Blob.fromData(data);
}
```

### Type Guard Pattern

```typescript
import { Blob, type BrandedBlob } from '@tevm/voltaire';

function assertBlob(value: Uint8Array): asserts value is BrandedBlob {
  if (!Blob.isValid(value)) {
    throw new Error(`Invalid blob size: ${value.length}`);
  }
}

// Usage
const data = new Uint8Array(131072);
assertBlob(data); // Throws if invalid
// data is now typed as BrandedBlob
```
