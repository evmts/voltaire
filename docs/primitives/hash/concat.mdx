## Implementation

Concatenates bytes and hashes result:

```typescript
function concat(...hashes: BrandedHash[]): BrandedHash {
  const totalLength = hashes.reduce((sum, hash) => sum + hash.length, 0)
  const combined = new Uint8Array(totalLength)

  let offset = 0
  for (const hash of hashes) {
    combined.set(hash, offset)
    offset += hash.length
  }

  return keccak256(combined)
}
```

## Usage Examples

### Merkle Tree Parent Nodes

```typescript
function hashPair(left: Hash, right: Hash): Hash {
  return Hash.concat(left, right)
}

// Build Merkle tree level
function buildLevel(leaves: Hash[]): Hash[] {
  const parents: Hash[] = []

  for (let i = 0; i < leaves.length; i += 2) {
    if (i + 1 < leaves.length) {
      parents.push(Hash.concat(leaves[i], leaves[i + 1]))
    } else {
      parents.push(leaves[i])
    }
  }

  return parents
}
```

### Commitment Schemes

```typescript
// Commit to multiple values
function createMultiCommitment(values: Hash[], salt: Hash): Hash {
  return Hash.concat(...values, salt)
}
```
