## Usage Examples

### Format Function

```typescript
import * as BrandedAbi from '@tevm/voltaire/Abi/BrandedAbi'

const fn = {
  type: "function",
  name: "balanceOf",
  inputs: [{ type: "address", name: "owner" }],
  outputs: [{ type: "uint256", name: "balance" }]
}

console.log(BrandedAbi.format(fn))
// "function balanceOf(address owner) returns (uint256 balance)"
```

### Format Event

```typescript
import * as BrandedAbi from '@tevm/voltaire/Abi/BrandedAbi'

const event = {
  type: "event",
  name: "Transfer",
  inputs: [
    { type: "address", name: "from", indexed: true },
    { type: "address", name: "to", indexed: true },
    { type: "uint256", name: "value" }
  ]
}

console.log(BrandedAbi.format(event))
// "event Transfer(address indexed from, address indexed to, uint256 value)"
```

### Format Error

```typescript
import * as BrandedAbi from '@tevm/voltaire/Abi/BrandedAbi'

const error = {
  type: "error",
  name: "InsufficientBalance",
  inputs: [
    { type: "uint256", name: "balance" },
    { type: "uint256", name: "required" }
  ]
}

console.log(BrandedAbi.format(error))
// "error InsufficientBalance(uint256 balance, uint256 required)"
```
