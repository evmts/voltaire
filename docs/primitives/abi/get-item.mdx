## Usage Examples

### Find and Use Function

```typescript
import * as BrandedAbi from '@tevm/voltaire/Abi/BrandedAbi'

const abi = [...] // Full contract ABI

const transferFn = BrandedAbi.getItem(abi, "transfer", "function")

if (transferFn && transferFn.type === "function") {
  const calldata = BrandedAbi.Function.encodeParams(transferFn, [to, amount])
  await provider.call({ to: contractAddress, data: calldata })
}
```

### Error Handling

```typescript
import * as BrandedAbi from '@tevm/voltaire/Abi/BrandedAbi'

const abi = [...]

const item = BrandedAbi.getItem(abi, "nonExistent", "function")
if (!item) {
  throw new Error("Function not found in ABI")
}
```
