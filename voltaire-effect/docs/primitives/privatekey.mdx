```typescript
import * as PrivateKey from 'voltaire-effect/primitives/PrivateKey'
import { Effect } from 'effect'
import * as S from 'effect/Schema'

// From hex (32 bytes)
const pk = S.decodeSync(PrivateKey.PrivateKeySchema)(
  '0x0123456789abcdef0123456789abcdef0123456789abcdef0123456789abcdef'
)

// From bytes
const pk2 = S.decodeSync(PrivateKey.PrivateKeyFromBytesSchema)(
  new Uint8Array(32).fill(1)
)
```

## Effect API

```typescript
// From hex
const pk = Effect.runSync(PrivateKey.from('0x0123...'))

// From bytes
const pk = Effect.runSync(PrivateKey.fromBytes(new Uint8Array(32).fill(1)))
```

## Errors

```typescript
const result = await Effect.runPromiseExit(PrivateKey.from('0xinvalid'))
// InvalidFormatError | InvalidLengthError
```
