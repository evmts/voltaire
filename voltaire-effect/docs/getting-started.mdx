```bash
pnpm add voltaire-effect @tevm/voltaire effect
```

## Schema Validation

```typescript
import { Address, Hex } from 'voltaire-effect'
import * as Schema from 'effect/Schema'

const addr = Schema.decodeSync(Address.Schema)("0x742d35Cc...")

// Returns actual Voltaire branded type - use directly
import { Address as VAddr } from '@tevm/voltaire'
VAddr.toChecksummed(addr)
```

## Effect Operations

```typescript
import { Address } from 'voltaire-effect'
import { Effect } from 'effect'

const program = Effect.gen(function* () {
  const addr = yield* Address.fromBytes(bytes)
  return Address.toChecksummed(addr)
})

Effect.runPromise(program)
```

## Error Handling

```typescript
import { Address } from 'voltaire-effect'
import { Effect } from 'effect'

Address.fromString("invalid").pipe(
  Effect.catchTag("InvalidAddress", (e) => 
    Effect.succeed("0x0000000000000000000000000000000000000000")
  )
)
```
